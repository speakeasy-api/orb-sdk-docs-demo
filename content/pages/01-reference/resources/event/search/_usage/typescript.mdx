import { Orb } from "orb-billing";

(async() => {
  const sdk = new Orb({
    security: {
      apiKeyAuth: "",
    },
  });

  const res = await sdk.event.search({
    eventSearchCriteria: {
      eventIds: [
        "string",
      ],
    },
  });

  if (res.statusCode == 200) {
    do {
      // handle items

      res = res.next();
    } while (res);
  }
})();
